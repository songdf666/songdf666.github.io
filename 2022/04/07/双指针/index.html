双指针算法


解：

#include<iostream>
using namespace std;
int n,a[100005],k,ans,q[100005],l;
int main(){
   cin>>n;
   for(int i=0;i<n;i++){
       cin>>a[i];
   }
   for(int i=0,j=0;i<n;i++){
       q[a[i]]++;
       while(q[a[i]]>1&&j<=i){
           q[a[j]]--;
           j++;
       }
       ans=max(ans,i-j+1);
   }
   cout<<ans;
   return 0;
}
